    background: "#222222",

  const changeTheme = () => {

  return (
    <>
      <ThemedButton />
    </>
  );
}

  const { theme, changeTheme } = useContext(ThemeContext);
  return (
    <button
      style={{ background: theme.background, color: theme.foreground }}
      onClick={changeTheme}
    >
      I am styled by theme context!
    </button>
  );
}
